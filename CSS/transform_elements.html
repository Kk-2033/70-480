Transform Elements

var Left = function(){
   $('#play').animate({left: "-=50px"}, 1000);
};
var Right = function(){
   $('#play').animate({left: "+=50px"}, 1000);
};
var Down = function(){
  $('#play').animate({top: "+=50px"}, 1000);
};
var Up = function(){
  $('#play').animate({top: "-=50px"}, 1000);
};
var Invisible = function(){
  $('#play').css({visibility: "hidden"});
}; 
var Show = function(){
  $('#play').css({visibility: "visible"});
};
