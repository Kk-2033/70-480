Canvas

function drawRect() {
  var canvas = document.getElementById('myCanvas');
  ctx = canvas.getContext('2d');
  ctx.fillStyle = 'yellow';
  ctx.strokeStyle = 'blue';
  ctx.lineWidth = 10;
  ctx.beginPath();
  ctx.moveTo(100, 100);
  ctx.lineTo(150, 50);
  ctx.lineTo(200, 100);
  ctx.rect(100, 100, 100, 100);
  ctx.fill();
  ctx.stroke();
  
  ctx.strokeStyle = 'red';
  ctx.lineWidth = 5;
  ctx.beginPath();
  ctx.rect(125, 125, 50, 50);
  ctx.stroke();
}
