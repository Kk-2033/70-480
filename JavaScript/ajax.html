var callService = function(){
  var payload = $('#queryValue').val();
  $.ajax({
    type: 'GET', 
    url: 'http://md5.jsontest.com/', 
    data: { text:  payload}, 
    dataType: 'json',
    success: function (data) { 	
	  $('body').append($('

      }));				
    }
  });
};
